 58        if base_key:
 59            # If a base_key is provided, use it and append a number if necessary

 62            while self.tool_cache.get(key) is not None:

 65        else:
 66            # If no base_key, use timestamp and random string

 89                self.dataframe = pd.read_json(file_path)

 95                self.dataframe = pd.read_parquet(file_path, parse_dates=True)
 96            else:
 97                return f"Unsupported file type: {file_extension}"
 98
 99            if self.dataframe is None:
100                return f'Error loading from file: {file_path}'

264    async def add_column(self, **kwargs):
265        column_name = kwargs.get('column_name')
266        column_value_or_expression = kwargs.get('column_value_or_expression')
267        is_expression = kwargs.get('is_expression', False)
268
269        if self.dataframe is None:
270            return 'No DataFrame is loaded. Please load a file first.'
271        print(
272            f'Adding column {column_name} with value {column_value_or_expression}. Expression flag is: {is_expression}')
273        try:
274            if is_expression:
275                try:
276                    eval_context = {
277                        'pd': pd,
278                        'np': np,
279                        'datetime': datetime,
280                        'dataframe': self.dataframe,
281                        'df': self.dataframe
282                    }
283
284                    # Add all column names to the context
285                    for col in self.dataframe.columns:
286                        eval_context[col] = self.dataframe[col]
287
288                    def replace_column_names(match):
289                        column_name = match.group(1)
290                        if column_name in self.dataframe.columns:
291                            return f"dataframe['{column_name}']"
292                        return match.group(0)
293
294                    # Preprocess the expression
295                    modified_value = column_value_or_expression.replace('df[', 'dataframe[')
296                    modified_value = modified_value.replace('df.', 'dataframe.')
297
298                    # Replace column names, but not within quotes or function calls
299                    modified_value = re.sub(r'(?<![\w\'"])(\w+)(?![\w\'"])', replace_column_names, modified_value)
300
301                    # Wrap the entire expression in parentheses to ensure it's evaluated as a whole
302                    modified_value = f"({modified_value})"
303
304                    result = eval(modified_value, eval_context)
305
306                    if isinstance(result, pd.Series):
307                        self.dataframe[column_name] = result
308                    elif isinstance(result, pd.Timedelta):
309                        self.dataframe[column_name] = result.total_seconds() / (24 * 60 * 60)  # Convert to days
310                    else:
311                        self.dataframe[column_name] = pd.Series(result, index=self.dataframe.index)
312
313                    return f'Column {column_name} added successfully using custom eval.'
314                except Exception as e:
315                    return f'Error adding column: Custom eval failed. Error: "{str(e)}"'
316            else:
317                # If not an expression, just assign the value directly
318                self.dataframe[column_name] = column_value_or_expression
319                return self._standardized_result(
320                    message=f"Column {column_name} added successfully with direct value assignment")
321        except Exception as e:
322            print(f'Error adding column: {str(e)}')
323            logger.error(f'Error adding column: {str(e)}')
324            return f'Error adding column: {str(e)}'
