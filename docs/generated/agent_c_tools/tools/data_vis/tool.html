 23    async def _load_data(self, in_memory_or_cache: str, data_key: str = None) -> pd.DataFrame:
 24        if in_memory_or_cache == 'memory':
 25            df = self.dataframe_tool.dataframe
 26            if df is None:
 27                raise ValueError("No data available in memory. Please load data first.")
 28        elif in_memory_or_cache == 'cache':
 29            if data_key is None:
 30                raise ValueError("data_key must be provided when loading from cache")
 31
 32            data = self.tool_cache.get(data_key)
 33            if data is None:
 34                raise ValueError(f"No data found in tool_cache for key: {data_key}")
 35
 36            if isinstance(data, bytes):
 37                # It's parquet data, read it into a DataFrame
 38                df = pd.read_parquet(io.BytesIO(data))
 39            elif isinstance(data, dict):
 40                # It's extracted data for plotting
 41                df = pd.DataFrame(data)
 42            else:
 43                raise ValueError(f"Unknown data format in tool_cache for key: {data_key}")
 44        else:
 45            raise ValueError("Invalid value for in_memory_or_cache. Must be 'memory' or 'cache'.")
 46
 47        return df

 78        if value >= 1e9:

135    async def create_bar_chart(self, **kwargs):
136        try:
137            df = await self._load_data(in_memory_or_cache=kwargs['in_memory_or_cache'], data_key=kwargs.get('data_key'))
138            plt.figure(figsize=(12, 6))
139            sns.set_style("whitegrid")
140
141            x_column = kwargs['x_column']
142            y_columns = kwargs['y_columns']
143            secondary_y = kwargs.get('secondary_y', [])
144            show_data_labels = kwargs.get('show_data_labels', False)
145
146            # Check if x_column is a date
147            is_date_axis = False
148            try:
149                df[x_column] = pd.to_datetime(df[x_column])
150                is_date_axis = True
151            except ValueError:
152                pass  # Not a date column, proceed with normal plotting
153
154            # Melt the dataframe for seaborn
155            df_melted = df.melt(id_vars=[x_column], value_vars=y_columns, var_name='Variable', value_name='Value')
156
157            # Create the main plot
158            ax = sns.barplot(x=x_column, y='Value', hue='Variable', data=df_melted)
159
160            # Handle secondary y-axis
161            if secondary_y:
162                ax2 = ax.twinx()
163                df_secondary = df_melted[df_melted['Variable'].isin(secondary_y)]
164                sns.barplot(x=x_column, y='Value', hue='Variable', data=df_secondary, ax=ax2, alpha=0.5)
165                ax2.grid(False)  # Turn off grid for secondary axis
166
167            ax.yaxis.set_major_formatter(FuncFormatter(self._format_large_number))
168            if secondary_y:
169                ax2.yaxis.set_major_formatter(FuncFormatter(self._format_large_number))
170
171            # Set labels and title
172            plt.xlabel(kwargs.get('x_label', x_column))
173            ax.set_ylabel(kwargs.get('y_label', ', '.join([col for col in y_columns if col not in secondary_y])))
174            if secondary_y:
175                ax2.set_ylabel(', '.join(secondary_y))
176            plt.title(kwargs.get('title', f'Bar Chart of {", ".join(y_columns)} by {x_column}'))
177
178            # Label x-axis and if necessary handle dates
179            if is_date_axis:
180                self._setup_date_axis(ax, df, x_column, kwargs.get('x_date_format', '%Y-%m-%d'),
181                                      kwargs.get('x_date_locator', 'auto'))
182            else:
183                plt.xticks(rotation=45, ha='right')
184
185            # Add data labels if requested
186            if show_data_labels:
187                for p in ax.patches:
188                    height = p.get_height()
189                    ax.annotate(self._format_large_number(height, 0),
190                                (p.get_x() + p.get_width() / 2., height),
191                                ha='center', va='bottom',
192                                xytext=(0, 5), textcoords='offset points')
193            # Adjust legend
194            handles, labels = ax.get_legend_handles_labels()
195            if secondary_y:
196                handles2, labels2 = ax2.get_legend_handles_labels()
197                handles += handles2
198                labels += labels2
199            plt.legend(handles, labels, loc='upper left', bbox_to_anchor=(1, 1))
200
201            # Adjust layout and save
202            plt.tight_layout()
203
204            result = await self._save_plot(plt.gcf(), kwargs['output_filename'], kwargs.get('workspace_name', 'project'))
205            return json.dumps(result)
206        except Exception as e:
207            return json.dumps({"error": f"Error creating bar chart: {str(e)}"})

371        try:
372            df = await self._load_data(in_memory_or_cache=kwargs['in_memory_or_cache'], data_key=kwargs.get('data_key'))
373            columns = kwargs['columns']
374            bins = kwargs.get('bins', 30)
375            kde = kwargs.get('kde', False)
376
377            # Calculate number of rows and columns for subplots
378            n_plots = len(columns)
379            n_rows = (n_plots + 1) // 2  # 2 columns of subplots
380            n_cols = min(n_plots, 2)
381
382            figsize = kwargs.get('figsize', (12, 6 * n_rows))
383            fig, axes = plt.subplots(n_rows, n_cols, figsize=figsize)
384            if n_plots == 1:
385                axes = np.array([axes])
386            axes = axes.flatten()
387
388            sns.set_style("whitegrid")
389            colors = plt.cm.tab10(np.linspace(0, 1, n_plots))
390
391            bin_info_dict = {}
392
393            for i, column in enumerate(columns):
394                ax = axes[i]
395                sns.histplot(data=df, x=column, bins=bins, kde=kde, ax=ax, color=colors[i])
396
397                counts, bin_edges = np.histogram(df[column], bins=bins)
398                bin_info = [{"bin_range": f"{bin_edges[j]:.2f}-{bin_edges[j + 1]:.2f}", "count": int(count)}
399                            for j, count in enumerate(counts)]
400                bin_info_dict[column] = bin_info
401
402                ax.set_title(column)
403                ax.set_xlabel('Value')
404                ax.set_ylabel('Frequency')
405
406            # Remove any unused subplots
407            for i in range(n_plots, len(axes)):
408                fig.delaxes(axes[i])
409
410            plt.tight_layout()
411            fig.suptitle(kwargs.get('title', 'Multi-Column Histogram Plot'), fontsize=16, y=1.02)
412

414
415            # Add bin information to the result
