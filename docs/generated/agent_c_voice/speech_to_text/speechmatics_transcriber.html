236    def __handle_final_transcript(self, msg):
237        """
238        Event handler to send the transcript to the server
239
240        Args:
241            msg (dict): The transcript message received.
242        """
243        if len(msg['metadata']['transcript'].strip()) > 0:
244            lines = []
245            active_speaker = None
246            current_line: Union[RTTranscriptLine, None] = None
247
248            for result in msg['results']:
249                if self.diarization == 'speaker':
250                    msg_speaker = result["alternatives"][0]["speaker"]
251                else:
252                    msg_speaker = 'UU'
253
254                # It the speaker changes we need to start a new line of dialog
255                if active_speaker != msg_speaker:
256                    if current_line is not None:
257                        lines.append(current_line)
258
259                    current_line = RTTranscriptLine(speaker=msg_speaker, content='', start_time=msg['metadata']['start_time'])
260                    active_speaker = msg_speaker
261
262                token = result["alternatives"][0]["content"]
263                if len(token) > 0:
264                    current_line.end_time = msg['metadata']['end_time']
265
266                    # Punctuation tokens should not have a space preceding them
267                    if result["type"] == 'punctuation':
268                        current_line.content = current_line.content.strip() + token
269
270                        # and if it ends a sentence we should start a new line of dialog.
271                        if token in END_SENTENCE_PUNCTUATION:
272                            lines.append(current_line)
273                            current_line = None
274                            active_speaker = None
275                    else:
276                        if current_line.content.endswith(' ') or current_line.content == '' or token[0] in " .!?;:'\"":
277                            current_line.content += token
278                        else:
279                            current_line.content += f" {token}"
280
281            if current_line is not None and current_line.content.strip() != '':
282                self.empty_finals = 0
283                lines.append(current_line)
284            else:
285                self.empty_finals += 1
286
287            if len(lines) > 0:
288                # Don't sound a frame grab for the empty finals or the last punctuation token
289                if self._last_frame is not None and (len(lines) > 1 or len(lines[0].content) > 2):
290                    self._play_sound("frame_grab")
291
292                self._publish_final_transcript(lines, self._last_frame)
293                self._last_frame = None
294
295            if self.empty_finals > 0 or self.input_complete_event.is_set() or self.shutdown_event.is_set():
296                self._publish_pause(self.empty_finals >= self.max_empty_finals)

298    def _run(self):
299        """
300        Runs the real-time transcription.
301        """
302        try:
303            self.logger.debug("Speechmatics connection Opened.")
304            self.sm_web_socket.run_synchronously(self.audio_processor, self.trans_conf, self.sm_settings)
305            self.logger.debug("Speechmatics connection closed.")
306        except KeyboardInterrupt:
307            return
308        except HTTPStatusError as e:
309            if e.response.status_code == 401:
310                print('Invalid Speechmatics API key. Transcription stopped.')
311                return
312            else:
313                raise e

347    def start(self):
348        if self.thread is not None and self.shutdown_event.is_set():
349            self.thread.join()
350            self.thread = None
351            self.shutdown_event.clear()
352
353        if self.thread is None or not self.thread.is_alive():
354            self.logger.debug("Starting transcription thread.")
355            self.thread = threading.Thread(target=self._run, daemon=True)
356            self.thread.start()
357
358        self.input_complete_event.clear()
359        return self.thread
360
361    def stop(self):
362        if self.thread is not None and self.thread.is_alive():
363            self.logger.debug("Stopping transcription thread.")
364            self.listen_stop()
365            self.shutdown_event.set()
366            self.thread.join()
367            self.thread = None
368
369    def shutdown(self):
370        if self.thread is not None and self.thread.is_alive():
371            self.logger.debug("Stopping transcription thread.")
372            self.listen_stop()
373            self.shutdown_event.set()

237    def __handle_final_transcript(self, msg):
238        """
239        Event handler to send the transcript to the server
240
241        Args:
242            msg (dict): The transcript message received.
243        """
244        if len(msg['metadata']['transcript'].strip()) > 0:
245            lines = []
246            active_speaker = None
247            current_line: Union[RTTranscriptLine, None] = None
248
249            for result in msg['results']:
250                if self.diarization == 'speaker':
251                    msg_speaker = result["alternatives"][0]["speaker"]
252                else:
253                    msg_speaker = 'UU'
254
255                # It the speaker changes we need to start a new line of dialog
256                if active_speaker != msg_speaker:
257                    if current_line is not None:
258                        lines.append(current_line)
259
260                    current_line = RTTranscriptLine(speaker=msg_speaker, content='', start_time=msg['metadata']['start_time'])
261                    active_speaker = msg_speaker
262
263                token = result["alternatives"][0]["content"]
264                if len(token) > 0:
265                    current_line.end_time = msg['metadata']['end_time']
266
267                    # Punctuation tokens should not have a space preceding them
268                    if result["type"] == 'punctuation':
269                        current_line.content = current_line.content.strip() + token
270
271                        # and if it ends a sentence we should start a new line of dialog.
272                        if token in END_SENTENCE_PUNCTUATION:
273                            lines.append(current_line)
274                            current_line = None
275                            active_speaker = None
276                    else:
277                        if current_line.content.endswith(' ') or current_line.content == '' or token[0] in " .!?;:'\"":
278                            current_line.content += token
279                        else:
280                            current_line.content += f" {token}"
281
282            if current_line is not None and current_line.content.strip() != '':
283                self.empty_finals = 0
284                lines.append(current_line)
285            else:
286                self.empty_finals += 1
287
288            if len(lines) > 0:
289                # Don't sound a frame grab for the empty finals or the last punctuation token
290                if self._last_frame is not None and (len(lines) > 1 or len(lines[0].content) > 2):
291                    self._play_sound("frame_grab")
292
293                self._publish_final_transcript(lines, self._last_frame)
294                self._last_frame = None
295
296            if self.empty_finals > 0 or self.input_complete_event.is_set() or self.shutdown_event.is_set():
297                self._publish_pause(self.empty_finals >= self.max_empty_finals)
298
299    def _run(self):
300        """
301        Runs the real-time transcription.
302        """
303        try:
304            self.logger.debug("Speechmatics connection Opened.")
305            self.sm_web_socket.run_synchronously(self.audio_processor, self.trans_conf, self.sm_settings)
306            self.logger.debug("Speechmatics connection closed.")
307        except KeyboardInterrupt:
308            return
309        except HTTPStatusError as e:
310            if e.response.status_code == 401:
311                print('Invalid Speechmatics API key. Transcription stopped.')
312                return
313            else:
314                raise e
