227    async def __init_gpt_chat_agent(self, persona_prompt: str):
228        """
229        This sets up the chat agent for the current session.
230        Here's where we declare the toolsets we want to use with the agent as well as how our system prompt will look.
231
232        Args:
233            persona_prompt (str): The prompt to initialize the ChatAgent with.
234        """
235        self.logger.debug("Initializing GPT Chat Agent...")
236        # Declare a "tool chest" of toolsets for the model.
237        # In this reference app we're not supplying a tool_classes parameter so it will grab the "kitchen sink"
238        # of all tool classes that have been registered.
239        self.tool_chest = ToolChest()
240
241        # These are the default and extra options for the various toolsets, since the toolsets all use kwargs
242        # we can send them the whole bag of options without worry.
243        tool_opts = {'tool_cache': self.tool_cache, 'session_manager': self.session_manager, 'user_preferences': self.user_prefs,
244                     'workspaces': self.workspaces, 'streaming_callback': self.chat_callback}
245
246        self.logger.debug("Initializing GPT Chat Agent... Initializing toolsets...")
247        await self.tool_chest.init_tools(**tool_opts)
248
249        # The prompt sections are grouped into "operational" and informational categories.
250        # Note the XML tag as the template for the Core Instruction template, by default the core instruction set template
251        # contains stuff that make talking to a development agent harder, so we shortcircuit it here BUT we have to have the XML tag there
252        # That tag anchors the critical parts of the prompt into a logical group that the model can "hold on to" easier.
253        # this also provides us a way to CLEARLY delineate between things WHICH MUST BE FOLLOWED and things that are informational
254        # or (more importantly) USER generated.  By establishing 'operating guidelines' as a thing we can make the model evaluate its output
255        # as well as user requests based on them with far greater accuracy and reliability.
256        self.logger.debug("Initializing GPT Chat Agent... Initializing sections...")
257        operating_sections = [
258            CoreInstructionSection(template="<operating_guidelines>\n"),
259            PersonaSection(template=persona_prompt),
260            EndOperatingGuideLinesSection()
261        ]
262
263        operating_sections.extend(self.tool_chest.active_tool_sections)
264
265        # Anything outside the operating guidelines is basically helpful information for the model to know.
266        # These are demo sections that tell the model a little about the user as well as things like the current date / time.
267        info_sections = [HelpfulInfoStartSection(),
268                         EnvironmentInfoSection(session_manager=self.session_manager, voice_tools=self.tool_chest.active_tools.get('voice')),
269                         UserBioSection(session_manager=self.session_manager)]
270
271        self.logger.debug("Initializing GPT Chat Agent... Initializing agent...")
272        # FINALLY we create the agent
273        self.agent: GPTChatAgent = GPTChatAgent(prompt_builder=PromptBuilder(sections=operating_sections + info_sections),
274                                                model_name=self.model_name,
275                                                tool_chest=self.tool_chest,
276                                                streaming_callback=self.chat_callback,
277                                                output_format=self.agent_output_format)

306        """

351    async def __core_input_loop(self):
352        """
353        This is the core loop for user input and chat interaction.
354
355        Continuously prompts input from the user and sends it to the ChatAgent, unless '!exit' command is entered
356        or an EOF(End-of-File)/KeyboardInterrupt exception occurs.
357
358        The chat responses and events are handled asynchronously by a callback.
359
360        If any other exception occurs, it is logged.
361        """
362        self.logger.debug("Starting core input loop...")
363
364        if self.voice_mode and self.tts_engine is not None:
365            voice_tool = self.tool_chest.active_tools.get('voice')
366            self.tts_engine.set_voice(voice_tool.voice, self.tts_model_id)
367
368        while not self.exit_event.is_set():
369            try:
370                user_message, frame = await self.chat_ui.get_user_input()
371                if self.tts_engine and self.tts_engine.tts_active.is_set():
372                    self.tts_engine.cancel()
373
374                if self.transcriber is not None:
375                    self.transcriber.shutdown()
376
377                if await self.cmd_handler.handle_command(user_message, self):
378                    continue
379
380                # We wait till the last minute to refresh out data from Zep so we can allow it to summarize and whatnot in the background
381                await self.session_manager.update()
382
383                # Here's where we send agent the input from the user.
384                # We give it a zep cache to construct a message array from if the current_chat_log is None.
385                # We als pass it a property bag to be used in prompt templates.
386
387                # The output of the gen_chat method is an async iterator that yields content tokens only.
388                # This iterator can then be passed to our async TTS engine to minimize the latency between
389                # when the tokens start coming in and the agent starts speaking.
390                #   agent_iter yields a token of text which gets consumed by the TTS engine
391                #   the TTS engine send the token up to ElevenLabs via a websocket whenever it gets a response, it yields that response
392                #   The response from that websocket is a chunk of MP3 audio, which we then stream to the audio player.
393                if frame is not None and self.agent.supports_multimodal:
394                    image_inputs = [frame.to_image_input()]
395                else:
396                    image_inputs = None
397
398                await self.agent.chat(session_manager=self.session_manager, user_message=user_message, prompt_metadata=await self.__build_prompt_metadata(),
399                                      messages=self.current_chat_Log, output_format=self.agent_output_format, images=image_inputs)
400
401                await self.session_manager.flush()
402            except (EOFError, KeyboardInterrupt):
403                # Handle exit upon EOF (Ctrl-D) or Keyboard Interrupt (Ctrl-C)
404                break
405            except Exception as e:
406                self.logger.exception(f"An error occurred: {e}")
407
408        if self.session_manager.is_new_session:
409            await self.session_manager.zep_client.memory.adelete_memory(self.session_manager.chat_session_id)
410

450def main():
451    parser = argparse.ArgumentParser(description="CLI Chat Interface")
452    parser.add_argument('--session', type=str, help='Existing session ID to use', default=None)
453    parser.add_argument('--model', type=str, help='The model name to use', default="gpt-4o")
454    parser.add_argument('--prompt_file', type=str, help='Path to a file containing the system prompt to use.', default='personas/default.md')
455    parser.add_argument('--userid', type=str, help='The userid for the session', default=os.environ.get("CLI_CHAT_USER_ID", "default"))
456    parser.add_argument('--voice', action='store_true', help='Enable text_iter to speech via ElevenLabs.  Requires ELEVEN_LABS_API_KEY be set.')
457    parser.add_argument('--oi', action='store_true', help='Enable experimental Open Interpreter support')
458    parser.add_argument('--vision', action='store_true', help='Enable vision support')
459    parser.add_argument('--camera', type=int, default=-1, help='What camera number to use for vision support.  Defaults to -1 which checks the env var, which defaults to 0.')
460    parser.add_argument('--claude', action='store_true', help='Use Claude as the agent instead of GPT-4-1106-preview.  This is experimental.')
461    args = parser.parse_args()
462    load_dotenv(override=True)
463
464    if args.prompt_file is not None:
465        try:
466            # Read file content into prompt
467            with open(args.prompt_file, 'r', encoding='utf-8') as file:
468                prompt = file.read()
469        except FileNotFoundError:
470            print(f"The file {args.prompt_file} was not found.")
471            exit(1)
472        except Exception as e:
473            print(f"An error occurred while reading the prompt file: {e}")
474            exit(1)
475
476    model: str = args.model
477    backend: str = 'openai'
478    if args.claude and args.model == 'gpt-4o':
479        model = 'claude-3-sonnet-20240229'
480
481    camera: int = args.camera
482    if args.vision:
483        if camera == -1:
484            camera = int(os.environ.get('VIDEO_CAPTURE_DEVICE_NUM', 0))
485
486    if args.claude:
487        backend = 'claude'
488
489    chat = CLIChat(user_id=args.userid, prompt=prompt, session_id=args.session, voice=args.voice, model_name=model, allow_oi=args.oi, backend=backend,
490                   camera_no=camera)
491    asyncio.run(chat.run())

228    async def __init_gpt_chat_agent(self, persona_prompt: str):
229        """
230        This sets up the chat agent for the current session.
231        Here's where we declare the toolsets we want to use with the agent as well as how our system prompt will look.
232
233        Args:
234            persona_prompt (str): The prompt to initialize the ChatAgent with.
235        """
236        self.logger.debug("Initializing GPT Chat Agent...")
237        # Declare a "tool chest" of toolsets for the model.
238        # In this reference app we're not supplying a tool_classes parameter so it will grab the "kitchen sink"
239        # of all tool classes that have been registered.
240        self.tool_chest = ToolChest()
241
242        # These are the default and extra options for the various toolsets, since the toolsets all use kwargs
243        # we can send them the whole bag of options without worry.
244        tool_opts = {'tool_cache': self.tool_cache, 'session_manager': self.session_manager, 'user_preferences': self.user_prefs,
245                     'workspaces': self.workspaces, 'streaming_callback': self.chat_callback}
246
247        self.logger.debug("Initializing GPT Chat Agent... Initializing toolsets...")
248        await self.tool_chest.init_tools(**tool_opts)
249
250        # The prompt sections are grouped into "operational" and informational categories.
251        # Note the XML tag as the template for the Core Instruction template, by default the core instruction set template
252        # contains stuff that make talking to a development agent harder, so we shortcircuit it here BUT we have to have the XML tag there
253        # That tag anchors the critical parts of the prompt into a logical group that the model can "hold on to" easier.
254        # this also provides us a way to CLEARLY delineate between things WHICH MUST BE FOLLOWED and things that are informational
255        # or (more importantly) USER generated.  By establishing 'operating guidelines' as a thing we can make the model evaluate its output
256        # as well as user requests based on them with far greater accuracy and reliability.
257        self.logger.debug("Initializing GPT Chat Agent... Initializing sections...")
258        operating_sections = [
259            CoreInstructionSection(template="<operating_guidelines>\n"),
260            PersonaSection(template=persona_prompt),
261            EndOperatingGuideLinesSection()
262        ]
263
264        operating_sections.extend(self.tool_chest.active_tool_sections)
265
266        # Anything outside the operating guidelines is basically helpful information for the model to know.
267        # These are demo sections that tell the model a little about the user as well as things like the current date / time.
268        info_sections = [HelpfulInfoStartSection(),
269                         EnvironmentInfoSection(session_manager=self.session_manager, voice_tools=self.tool_chest.active_tools.get('voice')),
270                         UserBioSection(session_manager=self.session_manager)]
271
272        self.logger.debug("Initializing GPT Chat Agent... Initializing agent...")
273        # FINALLY we create the agent
274        self.agent: GPTChatAgent = GPTChatAgent(prompt_builder=PromptBuilder(sections=operating_sections + info_sections),
275                                                model_name=self.model_name,
276                                                tool_chest=self.tool_chest,
277                                                streaming_callback=self.chat_callback,
278                                                output_format=self.agent_output_format)

352    async def __core_input_loop(self):
353        """
354        This is the core loop for user input and chat interaction.
355
356        Continuously prompts input from the user and sends it to the ChatAgent, unless '!exit' command is entered
357        or an EOF(End-of-File)/KeyboardInterrupt exception occurs.
358
359        The chat responses and events are handled asynchronously by a callback.
360
361        If any other exception occurs, it is logged.
362        """
363        self.logger.debug("Starting core input loop...")
364
365        if self.voice_mode and self.tts_engine is not None:
366            voice_tool = self.tool_chest.active_tools.get('voice')
367            self.tts_engine.set_voice(voice_tool.voice, self.tts_model_id)
368
369        while not self.exit_event.is_set():
370            try:
371                user_message, frame = await self.chat_ui.get_user_input()
372                if self.tts_engine and self.tts_engine.tts_active.is_set():
373                    self.tts_engine.cancel()
374
375                if self.transcriber is not None:
376                    self.transcriber.shutdown()
377
378                if await self.cmd_handler.handle_command(user_message, self):
379                    continue
380
381                # We wait till the last minute to refresh out data from Zep so we can allow it to summarize and whatnot in the background
382                await self.session_manager.update()
383
384                # Here's where we send agent the input from the user.
385                # We give it a zep cache to construct a message array from if the current_chat_log is None.
386                # We als pass it a property bag to be used in prompt templates.
387
388                # The output of the gen_chat method is an async iterator that yields content tokens only.
389                # This iterator can then be passed to our async TTS engine to minimize the latency between
390                # when the tokens start coming in and the agent starts speaking.
391                #   agent_iter yields a token of text which gets consumed by the TTS engine
392                #   the TTS engine send the token up to ElevenLabs via a websocket whenever it gets a response, it yields that response
393                #   The response from that websocket is a chunk of MP3 audio, which we then stream to the audio player.
394                if frame is not None and self.agent.supports_multimodal:
395                    image_inputs = [frame.to_image_input()]
396                else:
397                    image_inputs = None
398
399                await self.agent.chat(session_manager=self.session_manager, user_message=user_message, prompt_metadata=await self.__build_prompt_metadata(),
400                                      messages=self.current_chat_Log, output_format=self.agent_output_format, images=image_inputs)
401
402                await self.session_manager.flush()
403            except (EOFError, KeyboardInterrupt):
404                # Handle exit upon EOF (Ctrl-D) or Keyboard Interrupt (Ctrl-C)
405                break
406            except Exception as e:
407                self.logger.exception(f"An error occurred: {e}")
408
409        if self.session_manager.is_new_session:
410            await self.session_manager.zep_client.memory.adelete_memory(self.session_manager.chat_session_id)
