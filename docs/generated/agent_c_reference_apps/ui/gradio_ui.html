 60def get_available_tools():
 61    available_tools = []
 62    for tool_class in Toolset.tool_registry:
 63        tool_info = {
 64            'name': tool_class.__name__,
 65            'module': tool_class.__module__,
 66            'doc': tool_class.__doc__,
 67            'essential': tool_class.__name__ in ESSENTIAL_TOOLS

377    async def handle_gradio_input(self, history) -> AsyncGenerator[tuple[list, str], None]:
378        self.queue = asyncio.Queue()
379
380        async def producer():
381            try:
382                await self.session_manager.update()
383                image_inputs = self.image_inputs if self.image_inputs else None
384                handled = await self.cmd_handler.handle_command(self.user_message, self)
385                if not handled:
386                    self.debug_response = await self.agent.chat(
387                        session_manager=self.session_manager,
388                        user_message=self.user_message,
389                        prompt_metadata=await self.__build_prompt_metadata(),
390                        messages=self.current_chat_Log,
391                        output_format='markdown',
392                        images=image_inputs,
393                        temperature=self.temperature
394                    )
395
396                    if isinstance(self.debug_response, list):
397                        for chunk in self.debug_response:
398                            if isinstance(chunk, str):
399                                await self.queue.put(chunk)
400                            elif isinstance(chunk, dict):
401                                # Handle dictionary responses if needed
402                                pass
403                    elif isinstance(self.debug_response, str):
404                        await self.queue.put(self.debug_response)
405
406                await self.session_manager.flush()
407                self.user_message = ''
408                self.image_inputs = []
409            finally:
410                await self.queue.put(None)  # Signal completion
411
412        producer_task = asyncio.create_task(producer())
413        history[-1][1] = ""
414        self.partial_output = ""
415
416        try:
417            while True:
418                content = await self.queue.get()
419                if content is None:
420                    break
421                if isinstance(content, str):
422                    self.partial_output += content
423                    history[-1][1] = self.partial_output + self.tool_note
424                    # Yield history and None for debug_info_display
425                    yield history, None
426                self.queue.task_done()
427        finally:
428            await producer_task
429            await self.display_debug_info()
430            # After response is complete, yield history and updated debug_info
431            yield history, self.debug_info  # Yield the final outputs

433    async def display_debug_info(self):
434        if not ENHANCED_DEBUG_INFO:
435            self.debug_info = {
436                "info": "Enhanced Debug Info is currently turned off. Please set the environment variable ENHANCED_DEBUG_INFO=True to see additional debug information."
437            }
438            return
439
440        if not self.debug_response or not isinstance(self.debug_response, list):
441            self.debug_info = None
442            return
443        try:
444            # remove Message objects from response, so we're only looking at last call
445            dicts_only = [item for item in self.debug_response if isinstance(item, dict)]
446            # Extract relevant and correlate information
447            q_and_a = question_response(dicts_only)
448            tool_calls = align_tool_calls(dicts_only)
449            sys_prompt = system_prompt(dicts_only)
450            additional_payload = {'user_id': self.user_id, 'session_id': self.session_id,
451                                  'selected_tools': self.selected_tools,
452                                  'tool_cache_dir': self.tool_cache_dir, 'output_tool_arguments': ENHANCED_DEBUG_INFO,
453                                  'persona_name': self.persona_name, 'backend': self.backend,
454                                  'model_name': self.model_name}
455            self.debug_info = combine_debug_info([q_and_a, tool_calls, additional_payload, sys_prompt])
456            self.debug_response = []
457        except Exception as e:
458            self.logger.error(f"Failed to display debug information: {e}")
459            self.debug_info = [{'error': f"Failed to display debug information: {e}"}]
460        finally:
461            return

528    async def __init_gpt_chat_agent(self):
529        """
530        This sets up the chat agent for the current session.
531        Here's where we declare the toolsets we want to use with the agent as well as how our system prompt will look.
532
533        Args:
534            persona_prompt (str): The prompt to initialize the ChatAgent with.
535        """
536        self.logger.debug("Initializing GPT Chat Agent...")
537        # Declare a "tool chest" of toolsets for the model.
538        # In this reference app we're not supplying a tool_classes parameter so it will grab the "kitchen sink"
539        # of all tool classes that have been registered.
540        # self.tool_chest = ToolChest()
541        self.tool_chest = ToolChest(tool_classes=[
542            tool for tool in Toolset.tool_registry
543            if tool.__name__ in self.selected_tools
544        ])
545
546        # These are the default and extra options for the various toolsets, since the toolsets all use kwargs
547        # we can send them the whole bag of options without worry.
548        tool_opts = {'tool_cache': self.tool_cache, 'session_manager': self.session_manager, 'user_preferences': self.user_prefs,
549                     'workspaces': self.workspaces, 'streaming_callback': self.chat_callback}
550
551        self.logger.debug("Initializing GPT Chat Agent... Initializing toolsets...")
552        await self.tool_chest.init_tools(**tool_opts)
553
554        # The prompt sections are grouped into "operational" and informational categories.

644        """
645        Called by the ChatAgent and toolsets to notify us of events as they happen.
646        """
647        if event.role != self.last_role:

652                await self.queue.put(f"\n## From {event.role}:\n")
653            elif self.last_role != 'user':
654                await self.queue.put(f"<hr>")
655
656            self.last_role = event.role
657
658        if event.render_media is not None:
659            if 'image' in event.render_media.content_type:
660                if event.render_media.content is not None:
661                    src = f"data:{event.render_media.content_type};base64,{event.render_media.content}"
662                elif event.render_media.content_bytes is not None:
663                    src = f"data:{event.render_media.content_type};base64,{base64.b64encode(event.render_media.content_bytes).decode('utf-8')}"
664                else:
665                    src = event.render_media.url
666                if 'svg' in event.render_media.content_type:
667                    image_tag = f"<br>\n<img src=\"{src}\" >\n"
668                else:
669                    image_tag = f"<br>\n<img src=\"{src}\" style=\"max-width: 60%; height: auto;\" >\n"
670                await self.queue.put(image_tag)
671
672            return
673
674        if event.tool_use_active is not None:
675            if event.tool_use_active:

677                # TODO Env check

378    async def handle_gradio_input(self, history) -> AsyncGenerator[tuple[list, str], None]:
379        self.queue = asyncio.Queue()
380
381        async def producer():
382            try:
383                await self.session_manager.update()
384                image_inputs = self.image_inputs if self.image_inputs else None
385                handled = await self.cmd_handler.handle_command(self.user_message, self)
386                if not handled:
387                    self.debug_response = await self.agent.chat(
388                        session_manager=self.session_manager,
389                        user_message=self.user_message,
390                        prompt_metadata=await self.__build_prompt_metadata(),
391                        messages=self.current_chat_Log,
392                        output_format='markdown',
393                        images=image_inputs,
394                        temperature=self.temperature
395                    )
396
397                    if isinstance(self.debug_response, list):
398                        for chunk in self.debug_response:
399                            if isinstance(chunk, str):
400                                await self.queue.put(chunk)
401                            elif isinstance(chunk, dict):
402                                # Handle dictionary responses if needed
403                                pass
404                    elif isinstance(self.debug_response, str):
405                        await self.queue.put(self.debug_response)
406
407                await self.session_manager.flush()
408                self.user_message = ''
409                self.image_inputs = []
410            finally:
411                await self.queue.put(None)  # Signal completion
412
413        producer_task = asyncio.create_task(producer())
414        history[-1][1] = ""
415        self.partial_output = ""
416
417        try:
418            while True:
419                content = await self.queue.get()
420                if content is None:
421                    break
422                if isinstance(content, str):
423                    self.partial_output += content
424                    history[-1][1] = self.partial_output + self.tool_note
425                    # Yield history and None for debug_info_display
426                    yield history, None
427                self.queue.task_done()
428        finally:
429            await producer_task
430            await self.display_debug_info()
431            # After response is complete, yield history and updated debug_info
432            yield history, self.debug_info  # Yield the final outputs
