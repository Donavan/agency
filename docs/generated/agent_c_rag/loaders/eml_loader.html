274    def __extract_msg_meta(self, msg: EmailMessage) -> dict[str, Any]:
275        """Extracts metadata from an email message.
276
277        Args:
278            msg (Message): The email message object.
279
280        Returns:
281            dict[str, Any]: A dictionary containing the extracted metadata.
282        """
283        msg_data = {
284            'subject': msg['subject'],
285            'sender': msg['from'],
286            'date': msg['date'],
287            'body_text': ''
288        }
289
290        # Extract body text - only non-multipart or plain text parts
291        if msg.is_multipart():
292            for part in msg.walk():
293                content_type = part.get_content_type()
294                if content_type == 'text/plain' and not part.is_attachment():
295                    msg_data['body_text'] += part.get_content().strip()
296        else:
297            msg_data['body_text'] = msg.get_content().strip()
298
299        msg_data['body_text'] = self.__clean_body_text(msg_data['body_text'])
300
301
302        return msg_data

304    def __clean_body_text(self, body_text: str) -> str:
305        lines = body_text.replace("\u200c ", "").replace("\r\n", "\n").split("\n")
306        good_lines = []
307        websites = ['facebook', 'twitter', 'x.com', 'instagram', 'linkedin', 'youtube', 'pinterest', 'tiktok', 'snapchat', 'redatatech']
308        boilerplate = ['intended recipient', '________________________________', 'information security warning', 'this is an external email',
309                       '[cid:', 'sent from my iphone']
310
311        for line in lines:
312            ll = line.lower()
313            if self.contains_any_phrase(ll, websites + boilerplate):
314                continue
315            good_lines.append(line)
316
317        output = "\n".join(good_lines)
318        drop_symbols = ["\u200b", "\u202f"]
319
320
321        for symbol in drop_symbols:
322            output = output.replace(symbol, "")
323        return output

372        """
373        Flatten the email structure by appending the body text and attachments from nested messages.
374        """
375        body_text = msg_data.get('body_text', '')
376        attachments = msg_data.get('attachments', {})
377
378        # Check for nested messages
379        nested_msgs = msg_data.get('attachments', {}).get('message', [])
380        for nested_msg in nested_msgs:
381            temp_flattened = self.flatten_email(nested_msg['content'])
382            #nested_body_text = temp_flattened['body_text']
383            body_text += self.__forwarded_body_text(temp_flattened)
384
385            # Merge attachments from nested messages
386            nested_attachments = temp_flattened['attachments']
387            attachments = {**attachments, **nested_attachments}
388
389        msg_data['body_text'] = body_text
390        msg_data['attachments'] = attachments
391
392        return msg_data

305    def __clean_body_text(self, body_text: str) -> str:
306        lines = body_text.replace("\u200c ", "").replace("\r\n", "\n").split("\n")
307        good_lines = []
308        websites = ['facebook', 'twitter', 'x.com', 'instagram', 'linkedin', 'youtube', 'pinterest', 'tiktok', 'snapchat', 'redatatech']
309        boilerplate = ['intended recipient', '________________________________', 'information security warning', 'this is an external email',
310                       '[cid:', 'sent from my iphone']
311
312        for line in lines:
313            ll = line.lower()
314            if self.contains_any_phrase(ll, websites + boilerplate):
315                continue
316            good_lines.append(line)
317
318        output = "\n".join(good_lines)
319        drop_symbols = ["\u200b", "\u202f"]
320
321
322        for symbol in drop_symbols:
323            output = output.replace(symbol, "")
324        return output
